# 从一个数组中收集特定的数据

/**
 * 每隔两项收集、组装一次数据
 * ['09:00', '10:00', '11:00', '12:00'] =>
 * [{value:['09:00', '10:00']},{value:['12:00', '13:00']}]
 */
const source = ['09:00', '10:00', '11:00', '12:00'];

let collecter = [],
	temp = [];

source.forEach((data, index) => {
	temp.push(data);
	if (index % 2 === 1) {
		collecter.push({
			value: temp
		});
		temp = [];
	}
})

/**
 * 修改source，添加属性type
 * 要求：每隔两项收集一次数据，type为start所在对象内的time放置在前面，type为end放置在后面。
 * source => [{value:['09:00','10:00']},{value:['11:00','12:00']}]
 */
const source = [{
	type: 'start',
	time: '09:00'
}, {
	type: 'end',
	time: '10:00'
}, {
	type: 'end',
	time: '12:00'
}, {
	type: 'start',
	time: '11:00'
}];

let collecter = [],
	temp = [];//临时容器，用于存储时刻

/**
 * 现在,temp并不是按照顺序存放数据，而是根据type来存放，
 * type为start时，对应的time存放在第一位，type为end存放在第二位。
 */
source.forEach((data,index) => {
	const {
		type,
		time
	} = data;

	if (type === 'start') {
		temp.unshift(time);
	}else{
		temp.push(time);
	}

	//每隔两项，往collecter容器中push一次数据
	//假设每隔3项push一次，代码应调整为：index % 3 === 2
	//假设每隔4项push一次，代码应调整为：index % 4 === 3
	if (index % 2 === 1) {
		collecter.push({
			value: temp
		});
		temp = [];//清空临时容器
	}

})

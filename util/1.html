# 获取所有需要校验表单元素的key值

collectViewAttr3keys = (conditions) => {
	let keys = [];
	conditions && conditions.forEach(c => {
		c.items.forEach(item => {
			const {
				viewAttr,
				domkey
			} = item;

			if (viewAttr === 3) {
				let key;
				if (domkey.length > 1) {
					key = domkey.join('__');
				} else {
					key = domkey[0];
				}
				keys.push(key);
			}
		})
	})
	return keys
}

# 比较时间点

/**
 * 例：比较两个时间点11:00和12:00
 */
import moment from 'moment';

convertTimeToMoment = (time) => {
	const [hour, minute] = time.split(':');

	return moment().hour(Number(hour)).minute(Number(minute));
}

const fst = '11:00',
	scd = '12:00';

const fst_moment = this.convertTimeToMoment(fst),
	scd_moment = this.convertTimeToMoment(scd);

fst_moment.isBefore(scd_moment);//判断fst是否早于scd
scd_moment.isAfter(fst_moment);//判断scd是否晚于fst
fst_moment.isSame(scd_moment);//判断fst和scd是否相等
const minutes = Math.abs(fst_moment.diff(scd_moment, "m"))//获取fst和scd的时间差（以分钟计）

# 校验表单

validateForm = (form) => {
	return new Promise((resolve, reject) => {
		form.validateForm().then(f => {
			if (f.isValid) {
				resolve();
			} else {
				f.showErrors();
			}
		})
	})
}
